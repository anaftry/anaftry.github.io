DEBIAN WEB Server

Perintah:

apt install apache2

cd /var/www/html/

mkidr web1

mkdir web2

nano web1/index.html

index.html

web 2

nano web2/index.html

index.html

Step 1

cd /etc/apache2/sites-available

mv 000-default.conf backup

cp backup web1.conf

cp backup web2.conf

nano web1.conf

web1.conf 

        ServerName fitri.internal
        ServerAlias fitrioctavianalestari.my.id

        ServerAdmin webmaster@localhost
        DocumentRoot /var/www/html/web1


        ErrorLog ${APACHE_LOG_DIR}/error.log
        CustomLog ${APACHE_LOG_DIR}/access.log combined

Step 2

nano web2

web2.conf 

        ServerName biodata.fitri.internal
        ServerAlias biodata.fitrioctavianalestari.my.id

        ServerAdmin webmaster@localhost
        DocumentRoot /var/www/html/web2


        ErrorLog ${APACHE_LOG_DIR}/error.log
        CustomLog ${APACHE_LOG_DIR}/access.log combined

Step 3

a2ensite web1.conf

a2ensite web2.conf

Perintah Restart:

systemctl reload apache2

service apache2 status
