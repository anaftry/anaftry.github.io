DEBIAN SAMBA SERVER

Perintah:

apt install samba

mkdir -p /home/samba/guru

mkdir -p /home/samba/public

groupadd guru

useradd -m -s /bin/bash guru1

passwd guru1

usermod -aG guru guru1

chown -R root:guru /home/samba/guru

chmod -R 2770 /home/samba/guru

chmod -R 0777 /home/samba/public

smbpasswd -a guru1

nano /etc/samba/smb.conf

smb.conf 
[guru]
path = /home/samba/guru
browseable = yes
writable = yes
valid users = @guru
create mask = 0660
directory mask = 2770

[public]
path = /home/samba/public
browseable = yes
writable = yes
read only = no
guest ok = yes

Perintah Restart:

service samba restart

service samba status

Notes:

jika sudah berjala silahkan test saja dengan user guru1
